cid,cname,rid,rname,rloc,fid,fname

SELECT c.cid,c.cname,r.rid,r.rname,r.rloc,f.fid,f.fname
FROM faculty f
LEFT JOIN Courses c
ON f.fid =c.fid
JOIN room r ON r.rid=c.rid;
